# Firebase Storage Security Rules for Matrimony App
# Copy these rules to Firebase Console → Storage → Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Profile photos - allow anyone to read, authenticated users to upload their own
    match /profile_photos/{uid}.jpg {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == uid;
    }
  }
}
